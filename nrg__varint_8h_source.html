38 template<class T, class = typename enable_if<is_signed<T>::value>::type>
39 typename make_unsigned<T>::type varint_zigzag(T i){
40  return (i << 1) ^ (i >> numeric_limits<T>::digits);
41 }

43 template<class T, class = typename enable_if<is_unsigned<T>::value>::type>
44 typename make_signed<T>::type varint_zagzig(T i){
45  return (i >> 1) ^ (-(i & 1));
46 }

48 template<typename T>
49 typename enable_if<is_unsigned<T>::value, size_t>::type varint_decode(Packet& p, T& ans){
50  ans = 0;
51  size_t i = 0;
52  uint8_t byte = 0x80;
53 
54  while(i < varint_bytes<T>::value && (byte & 0x80)){
55  if(!p.remaining()) return 0;
56  p.read8(byte);
57  ans |= ((byte & 0x7F) << (i++ * 7));
58  }
59 
60  return i;
61 }
62 
63 template<typename T>
64 typename enable_if<is_unsigned<T>::value, size_t>::type varint_encode(Packet& p, T num){
65  uint8_t enc[varint_bytes<T>::value];
66  size_t i = 0;
67 
68  while(enc[i++] = (num & 0x7F), num >>= 7){
69  enc[i-1] |= 0x80;
70  }
71 
72  p.writeArray(enc, i);
73  return i;
74 }
